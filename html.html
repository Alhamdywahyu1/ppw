from pandas.api.types import CategoricalDtype

# Dapatkan urutan unik IP berdasarkan kemunculan pertama
unique_ips_ordered = df['Remote host'].drop_duplicates().tolist()

# Buat tipe kategori dengan urutan yang ditentukan
ip_category_type = CategoricalDtype(categories=unique_ips_ordered, ordered=True)

# Konversi kolom 'Remote host' ke tipe kategori
df['Remote host'] = df['Remote host'].astype(ip_category_type)

# Urutkan DataFrame berdasarkan 'Remote host' yang sekarang berupa kategori
df_sorted_by_first_ip = df.sort_values(by='Remote host')

print("DataFrame diurutkan berdasarkan 'Remote host' (IP address) berdasarkan kemunculan pertama. 5 baris pertama:")
